function saveProfile() { const name = document.getElementById('user-name').value; const userClass = document.getElementById('user-class').value; const userImg = document.getElementById('user-img').src; if(name && userClass) { try { localStorage.setItem('userName', name); localStorage.setItem('userClass', userClass); localStorage.setItem('userProfilePic', userImg); alert("नमस्ते " + name + ", आपकी जानकारी सुरक्षित कर ली गई है!"); window.location.href = "dashboard.html"; } catch (e) { alert("फोटो का साइज बहुत बड़ा है, कृपया दूसरी फोटो चुनें।"); } } else { alert("कृपया नाम और कक्षा दोनों भरें"); } }
